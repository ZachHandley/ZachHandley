```ts
// website/src/lib/flags.ts
export type Flags = {
  ab?: boolean;            // enable experiment
  maxDpr?: number;         // clamp DPR (e.g., 1.5)
  idleLoad3D?: boolean;    // defer 3D to idle/visible
  instancing?: boolean;    // instanced crates
  dynamicRes?: boolean;    // auto DPR under load
  meshopt?: boolean;       // GLTF MeshoptDecoder
  draco?: boolean;         // GLTF Draco
  ktx2?: boolean;          // Basis/KTX2 textures
  lowShadows?: boolean;    // smaller maps/fewer casters
  fx?: boolean;            // Fireball/effects toggle
  lod?: boolean;           // LOD for crates/dragon
  mobileTopN?: number;     // cap crates on small viewports
};
export function getFlags(): Flags {
  const url = new URLSearchParams(window.location.search);
  const fromUrl = Object.fromEntries(url.entries());
  const fromLs = JSON.parse(localStorage.getItem('threlteFlags') ?? '{}');
  const fromEnv = {
    ab: import.meta.env.PUBLIC_AB === '1',
    maxDpr: Number(import.meta.env.PUBLIC_MAX_DPR ?? '0') || undefined,
  };
  return { idleLoad3D: true, draco: true, ...fromEnv, ...fromLs, ...fromUrl };
}
```

**Crate/Link Placement Improvements**
Goals: avoid dragon occlusion, keep two-column balance, stable layout across resizes, adaptive density.

- **Occlusion mask**: Compute the dragon’s projected bounding box + margin; forbid placement inside it.
- **Poisson-disc sampling**: For each column region (left/right), fill with well-spaced positions that avoid the mask and maintain minimum spacing.
- **Adaptive density**: Fewer crates on small viewports; adjust `minDistance` and column widths based on viewport.
- **Snap-to-grid**: Round to a grid to keep visual stability; recalc only on breakpoint changes, not every resize pixel.

Code sketch (crateController.calculateLayout)
```ts
type Vec2 = [number, number];
interface LayoutFlags { density?: 'low'|'med'|'high'; avoidOccluder?: boolean; }
interface Rect { x: number; y: number; w: number; h: number; }

export function calculateLayout(view: { w: number; h: number }, dragonRect: Rect, crates: any[], flags: LayoutFlags) {
  const margin = 16;
  const occluder = flags.avoidOccluder !== false
    ? inflate(dragonRect, margin)
    : { x: -Infinity, y: -Infinity, w: 0, h: 0 };

  const density = flags.density ?? (view.w < 768 ? 'low' : 'med');
  const minDist = density === 'low' ? 64 : density === 'high' ? 32 : 48;
  const grid = 16;

  const leftRegion = { x: grid, y: grid, w: view.w * 0.4 - 2*grid, h: view.h - 2*grid };
  const rightRegion = { x: view.w * 0.6 + grid, y: grid, w: view.w * 0.4 - 2*grid, h: view.h - 2*grid };

  const left = poissonFill(leftRegion, minDist, occluder);
  const right = poissonFill(rightRegion, minDist, occluder);

  const positions = interleave(left, right).slice(0, crates.length)
    .map(p => snap(p, grid))
    .map((p, i) => ({ id: crates[i].id, pos: [p[0], crateHeight(crates[i]), p[1]] as [number, number, number] }));

  return positions;
}

function poissonFill(region: Rect, r: number, forbid: Rect): Vec2[] {
  const pts: Vec2[] = [];
  const active: Vec2[] = [];
  const k = 30;
  const inForbid = (x: number, y: number) => x >= forbid.x && x <= forbid.x + forbid.w && y >= forbid.y && y <= forbid.y + forbid.h;
  const dist2 = (a: Vec2, b: Vec2) => (a[0]-b[0])**2 + (a[1]-b[1])**2;

  const first: Vec2 = [rand(region.x, region.x+region.w), rand(region.y, region.y+region.h)];
  if (!inForbid(first[0], first[1])) { pts.push(first); active.push(first); }

  while (active.length) {
    const idx = (Math.random() * active.length) | 0;
    const center = active[idx];
    let found = false;
    for (let i = 0; i < k; i++) {
      const a = Math.random() * 2 * Math.PI;
      const m = rand(r, 2*r);
      const p: Vec2 = [center[0] + Math.cos(a) * m, center[1] + Math.sin(a) * m];
      if (p[0] < region.x || p[0] > region.x+region.w || p[1] < region.y || p[1] > region.y+region.h) continue;
      if (inForbid(p[0], p[1])) continue;
      if (pts.every(q => dist2(p, q) >= r*r)) { pts.push(p); active.push(p); found = true; break; }
    }
    if (!found) active.splice(idx, 1);
  }
  return pts;
}

const rand = (a: number, b: number) => a + Math.random() * (b - a);
const snap = (p: Vec2, g: number): Vec2 => [Math.round(p[0] / g) * g, Math.round(p[1] / g) * g];
const inflate = (r: Rect, m: number): Rect => ({ x: r.x - m, y: r.y - m, w: r.w + 2*m, h: r.h + 2*m });
const interleave = <T,>(a: T[], b: T[]) => a.reduce<T[]>((acc, v, i) => (acc.push(v, b[i]), acc), []).filter(Boolean as any);
```

export type Flags = {
  maxDPR?: number; dynamicDPR?: boolean; demandFrame?: boolean;
  lazyEffects?: boolean; gltfMeshopt?: boolean; gltfKtx2?: boolean; draco?: boolean;
  instancedCrates?: boolean; lodCrates?: boolean; mobileLite?: boolean;

const defaults: Flags = { maxDPR: 1.5, dynamicDPR: true, demandFrame: true, lazyEffects: true, draco: true, shadowPreset: 'low' };
export function resolveFlags(server?: Partial<Flags>) {
  const url = new URLSearchParams(location.search);
  const ls = JSON.parse(localStorage.getItem('flags') || '{}');
  const merged = { ...defaults, ...server, ...ls };

  localStorage.setItem('flags', JSON.stringify(merged));

}

// website/src/components/svelte/utils/crateController.svelte.ts
export type LayoutParams = {
  viewport: { w: number; h: number; aspect: number };
  dragon: { center: [number, number]; radius: number }; // in world XY plane
  leftCount: number; rightCount: number;
  flags: { layoutV2?: boolean; density?: number };      // density 0..1
};

export function calculatePositionsV2(p: LayoutParams) {
  if (!p.flags.layoutV2) return calculatePositionsV1(p); // current grid
  const margin = 0.1 * p.viewport.w;
  const rBase = lerp(0.12, 0.22, 1 - (p.flags.density ?? 0.5)); // sample radius
  const leftDomain = { minX: -p.viewport.w/2 + margin, maxX: 0 - margin };
  const rightDomain = { minX: 0 + margin, maxX: p.viewport.w/2 - margin };
  const bandY = { minY: -p.viewport.h/4, maxY: p.viewport.h/4 };

  const rejectDragon = (x: number, y: number) => {
    const dx = x - p.dragon.center[0], dy = y - p.dragon.center[1];
    return Math.hypot(dx, dy) < p.dragon.radius * 1.15; // safety factor
  };

  const left = poisson2D(leftDomain, bandY, rBase, p.leftCount, rejectDragon);
  const right = poisson2D(rightDomain, bandY, rBase, p.rightCount, rejectDragon);

  // Convert to 3D world (x,y -> x, z) at fixed Y height
  const to3D = (pt: [number, number]) => ({ x: pt[0], y: 0.5, z: pt[1] });
  return { left: left.map(to3D), right: right.map(to3D) };
}

// Fast Poisson-disc with k retries
function poisson2D(
  xRange: { minX: number; maxX: number }, yRange: { minY: number; maxY: number },
  r: number, target: number, reject?: (x: number, y: number) => boolean
): [number, number][] {
  const pts: [number, number][] = [];
  const r2 = r * r, k = 20;
  while (pts.length < target) {
    let placed = false, tries = 0;
    while (!placed && tries++ < k) {
      const x = rand(xRange.minX, xRange.maxX);
      const y = rand(yRange.minY, yRange.maxY);
      if (reject?.(x, y)) continue;
      let ok = true;
      for (const [px, py] of pts) if ((x - px)**2 + (y - py)**2 < r2) { ok = false; break; }
      if (ok) { pts.push([x, y]); placed = true; }
    }
    if (!placed) break; // filled domain
  }
  return pts;
}
