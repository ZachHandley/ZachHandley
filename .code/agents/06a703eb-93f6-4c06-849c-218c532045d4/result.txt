## Executive Summary

### Phase 2: Server-Side Infrastructure

#### Step 2.1: Create Appwrite Server Helper
**New File**: `website/src/lib/appwriteServer.ts`
```typescript
import { Client, Databases, Account, Query, type Models } from 'node-appwrite';
import { env } from 'astro:env/server';
import type { Link } from '~/types/baseSchemas';

class AppwriteServer {
  private client: Client;
  private databases: Databases;
  private account: Account;
  
  constructor() {
    this.client = new Client()
      .setEndpoint(env.APPWRITE_ENDPOINT)
      .setProject(env.APPWRITE_PROJECT_ID)
      .setKey(env.APPWRITE_API_KEY);
    
    this.databases = new Databases(this.client);
    this.account = new Account(this.client);
  }

  // Transform Appwrite document to Link type
  private documentToLink(doc: Models.Document): Link {
    return {
      id: doc.$id,
      name: doc.name,
      url: doc.url,
      type: doc.type,
      category: doc.category,
      icon: doc.icon,
      active: doc.active,
      order: doc.order
    };
  }

  async getPublishedLinks(): Promise<Link[]> {
    const response = await this.databases.listDocuments(
      'default', // database ID
      'links',
      [
        Query.equal('active', true),
        Query.orderAsc('order'),
        Query.limit(100)
      ]
    );
    return response.documents.map(doc => this.documentToLink(doc));
  }

  async getAllLinks(): Promise<Link[]> {
    const response = await this.databases.listDocuments(
      'default',
      'links',
      [
        Query.orderAsc('order'),
        Query.limit(100)
      ]
    );
    return response.documents.map(doc => this.documentToLink(doc));
  }

  async toggleLinkActive(linkId: string): Promise<void> {
    const doc = await this.databases.getDocument('default', 'links', linkId);
    await this.databases.updateDocument(
      'default',
      'links',
      linkId,
      { active: !doc.active }
    );
  }

  async updateLinkOrder(linkId: string, order: number): Promise<void> {
    await this.databases.updateDocument(
      'default',
      'links',
      linkId,
      { order }
    );
  }

  async createLink(link: Partial<Link>): Promise<void> {
    await this.databases.createDocument(
      'default',
      'links',
      'unique()',
      link
    );
  }

  async updateLink(linkId: string, updates: Partial<Link>): Promise<void> {
    await this.databases.updateDocument(
      'default',
      'links',
      linkId,
      updates
    );
  }

  async deleteLink(linkId: string): Promise<void> {
    await this.databases.deleteDocument('default', 'links', linkId);
  }

  // Admin authentication helpers
  async createAdminSession(email: string, password: string): Promise<Models.Session> {
    const session = await this.account.createEmailPasswordSession(email, password);
    return session;
  }

  async verifyAdminSession(sessionId: string): Promise<boolean> {
    try {
      const session = await this.account.getSession(sessionId);
      // Check if user has admin label
      const user = await this.account.get();
      return user.labels?.includes('admin') ?? false;
    } catch {
      return false;
    }
  }

  async deleteSession(sessionId: string): Promise<void> {
    await this.account.deleteSession(sessionId);
  }
}

export const appwriteServer = new AppwriteServer();
```

### Phase 3: Authentication Infrastructure

#### Step 3.1: Admin Login Endpoint
**New File**: `website/src/pages/api/auth/login.ts`
```typescript
import type { APIRoute } from 'astro';
import { appwriteServer } from '~/lib/appwriteServer';

export const POST: APIRoute = async ({ request, cookies }) => {
  try {
    const { email, password } = await request.json();
    
    const session = await appwriteServer.createAdminSession(email, password);
    
    // Set HttpOnly cookie with session
    cookies.set('admin_session', session.$id, {
      httpOnly: true,
      secure: true,
      sameSite: 'strict',
      maxAge: 60 * 60 * 24 * 7, // 7 days
      path: '/'
    });
    
    return new Response(JSON.stringify({ success: true }), {
      status: 200,
      headers: { 'Content-Type': 'application/json' }
    });
  } catch (error) {
    return new Response(JSON.stringify({ error: 'Invalid credentials' }), {
      status: 401,
      headers: { 'Content-Type': 'application/json' }
    });
  }
};
```

import type { APIRoute } from 'astro';
import { appwriteServer } from '~/lib/appwriteServer';

export const POST: APIRoute = async ({ cookies }) => {
  const sessionId = cookies.get('admin_session')?.value;
  
  if (sessionId) {
    try {
      await appwriteServer.deleteSession(sessionId);
    } catch {
      // Session might already be invalid
    }
    cookies.delete('admin_session', { path: '/' });
  }
  
  return new Response(JSON.stringify({ success: true }), {
    status: 200,
    headers: { 'Content-Type': 'application/json' }
  });
};

import type { AstroCookies } from 'astro';
import { appwriteServer } from './appwriteServer';

export async function requireAdmin(cookies: AstroCookies): Promise<boolean> {
  const sessionId = cookies.get('admin_session')?.value;
  
  if (!sessionId) return false;
  
  return await appwriteServer.verifyAdminSession(sessionId);
}
